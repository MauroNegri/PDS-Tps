datos = load('te.txt');
fm = 11025;   ## frecuencia de muestreo, dato del enunciado
x = datos';   ## convertir a vector fila

## t es el eje de tiempo en segundos de cada muestra.
## La muestra 1 ocurre en t=0, la muestra 2 en t=1/fm, etc.
## (0:length(x)-1) genera los indices 0,1,2,...,N-1
## al dividir por fm cada indice se convierte en segundos.
## Se arranca desde 0 y no desde 1 porque el tiempo fisico
## de la primera muestra es t=0, no t=1/fm.
t = (0:length(x)-1) / fm;

## -------------------------------------------------------
## PASO 2: Calcular la energía en ventanas cortas
## -------------------------------------------------------

## Para detectar donde hay tono y donde solo hay ruido,
## calculo la energía de la señal en pedazos pequeños.
## Si hay tono, la energía es alta. Si hay solo ruido, es baja.

## Por que ventana de 20ms:
##   Si la ventana es muy corta (ej 1ms ~ 11 muestras), la estimacion
##   de energía queda muy ruidosa porque hay pocas muestras.
##   Si es muy larga (ej 500ms), pierdo resolución temporal y puedo
##   mezclar el final de un digito con el inicio del siguiente, que
##   estan separados por apenas ~200-300ms.
##   20ms es un valor estandar en procesamiento de audio: es corto
##   suficiente para tener buena resolucion temporal, y largo suficiente
##   para estimar la energía con estabilidad estadistica.
##   En muestras: round(0.02 * 11025) = 220 muestras.
ventana = round(0.02 * fm);

## Por que paso de 10ms:
##   El paso define cada cuanto avanzo la ventana.
##   Con paso = ventana/2 = 10ms tengo un solapamiento del 50%.
##   Esto es convencional: da buena resolucion temporal sin calcular
##   de mas. Con paso menor tendria mas puntos pero informacion
##   muy redundante. Con paso mayor (ej igual a la ventana, sin overlap)
##   podria perderme transiciones que ocurren justo entre dos ventanas.
paso = round(0.01 * fm);

## Cuantas ventanas entran en la señal:
##   La primera ventana empieza en la muestra 1 y termina en ventana.
##   La segunda empieza en 1+paso, la tercera en 1+2*paso, etc.
##   La ultima ventana que entra completa es la que empieza en
##   1 + (n-1)*paso y termina en 1 + (n-1)*paso + ventana - 1 <= length(x)
##   Despejando n: n <= (length(x) - ventana)/paso + 1
##   El floor asegura quedarse con ventanas completas, sin pasarse.
n_ventanas = floor((length(x) - ventana) / paso) + 1;

energia   = zeros(1, n_ventanas);   ## pre-aloco para eficiencia
t_energia = zeros(1, n_ventanas);

for i = 1:n_ventanas
    ## Indice de inicio de la ventana i:
    ##   i=1 -> idx_ini = 0*paso + 1 = 1  (primera muestra)
    ##   i=2 -> idx_ini = 1*paso + 1 = paso+1
    ##   i=k -> idx_ini = (k-1)*paso + 1
    idx_ini = (i-1)*paso + 1;

    ## Indice de fin: avanza (ventana-1) muestras desde el inicio.
    ## Es ventana-1 y no ventana porque idx_ini ya cuenta como
    ## la primera muestra de la ventana.
    ## Ejemplo: ventana=4, idx_ini=1 -> idx_fin=4 -> muestras 1,2,3,4 (4 en total)
    idx_fin = idx_ini + ventana - 1;

    seg = x(idx_ini:idx_fin);

    ## Energía normalizada: sum(x²)/N
    ## Divido por el tamaño de la ventana para que la energía no dependa
    ## de cuantas muestras tiene la ventana. Si no normalizara, ventanas
    ## mas largas siempre tendrian energía mas alta solo por tener mas
    ## muestras, lo cual no refleja la intensidad real de la señal.
    energia(i) = sum(seg.^2) / ventana;

    ## Tiempo del centro de la ventana:
    ##   El centro está en la muestra idx_ini + ventana/2
    ##   Divido por fm para pasar a segundos.
    ##   Uso el centro y no el inicio porque representa mejor
    ##   el instante al que corresponde la energía calculada.
    t_energia(i) = (idx_ini + ventana/2) / fm;
end

## Imprimo estadisticas para entender la distribucion de la energia.
## La mediana es mas representativa que la media para estimar el nivel
## de ruido, porque no la afectan los picos de los tonos.
## Como la señal tiene mas muestras de ruido que de tono (los digitos
## duran ~0.5s de un total de 10s), la mediana cae en la zona de ruido.
fprintf('Energía máxima:  %.4f\n', max(energia));
fprintf('Energía mínima:  %.4f\n', min(energia));
fprintf('Energía media:   %.4f\n', mean(energia));
fprintf('Energía mediana: %.4f\n', median(energia));

## -------------------------------------------------------
## PASO 3: Elegir el umbral
## -------------------------------------------------------

## Para saber qué umbral usar, pruebo varios valores y cuento
## cuántos segmentos detecta cada uno. Como sé que son 7 dígitos,
## busco el umbral que de exactamente 7.
##
## La logica para contar segmentos es detectar transiciones 0->1
## en el vector binario (energia > umbral): cada vez que pasa de
## inactivo a activo, empezó un nuevo digito.

fprintf('\n Umbral   | Segmentos\n');
for u = [0.06, 0.07, 0.08, 0.09, 0.10, 0.12, 0.15]
    act = energia > u;
    cnt = 0;
    for i = 2:length(act)
        ## Comparo elemento i con el anterior i-1.
        ## Arranco desde i=2 porque i-1 necesita al menos el indice 1.
        if act(i)==1 && act(i-1)==0
            cnt = cnt + 1;
        end
    end
    fprintf('  %.2f    |  %d\n', u, cnt);
end
## Resultado del barrido:
##   0.06 -> 101 segmentos: demasiado bajo, el ruido activa el umbral
##   0.07 -> 40  segmentos: sigue detectando ruido como señal
##   0.08 -> 9   segmentos: casi, pero fragmenta algun digito
##   0.09 -> 7   segmentos: correcto
##   0.10 -> 7   segmentos: tambien funciona
##   0.12 -> 7   segmentos: tambien funciona
##   0.15 -> 12  segmentos: demasiado alto, parte digitos en pedazos
##
## Elijo 0.09 en lugar de 0.10 o 0.12 para tener mas margen hacia abajo:
## si algun digito tiene energia un poco mas baja de lo esperado, con
## 0.09 sigue siendo detectado. Con 0.12 podria perderse.

umbral = 0.09;
figure(2); hold on;
plot([t_energia(1) t_energia(end)], [umbral umbral], 'r--', 'LineWidth', 1.5);
legend('energía', 'umbral = 0.09');

## -------------------------------------------------------
## PASO 4: Detectar inicio y fin de cada dígito
## -------------------------------------------------------

## Convierto la energía en una señal binaria: 1 donde hay tono, 0 donde no.
activo = energia > umbral;

## Recorro buscando transiciones:
##   activo(i-1)=0 y activo(i)=1 -> empieza un digito, guardo i como inicio
##   activo(i-1)=1 y activo(i)=0 -> termina un digito, guardo i como fin
inicios = [];
fines   = [];
for i = 2:length(activo)
    if activo(i)==1 && activo(i-1)==0
        inicios(end+1) = i;
    elseif activo(i)==0 && activo(i-1)==1
        fines(end+1) = i;
    end
end

## Si la señal termina mientras hay un digito activo, no se detecta
## el fin. En ese caso agrego el ultimo indice como fin del ultimo digito.
if length(fines) < length(inicios)
    fines(end+1) = length(activo);
end

## Hasta aca, inicios y fines son indices de ventana, no de muestra.
## La ventana i empieza en la muestra (i-1)*paso + 1.
## Por eso para convertir: muestra_inicio = (indice_ventana - 1)*paso + 1
##
## Para fines uso el ultimo indice de la ventana, que es:
## muestra_inicio_ventana + ventana - 1 = (i-1)*paso + ventana
## El min(..., length(x)) evita pasarse del largo de la señal
## en el caso del ultimo digito.
inicios_m = (inicios - 1) * paso + 1;
fines_m   = min((fines - 1) * paso + ventana, length(x));

fprintf('\nSegmentos detectados: %d\n', length(inicios));
for k = 1:length(inicios)
    dur = (fines_m(k) - inicios_m(k)) / fm;
    fprintf('  Dígito %d: %.3f s - %.3f s  (%.3f s)\n', ...
            k, inicios_m(k)/fm, fines_m(k)/fm, dur);
end

## -------------------------------------------------------
## PASO 6: Graficar cada dígito por separado
## -------------------------------------------------------

figure(4); clf;
for k = 1:length(inicios)
    seg     = x(inicios_m(k):fines_m(k));
    ## Eje de tiempo local que arranca en 0 para cada digito.
    ## Util para ver la duración individual sin importar en qué
    ## momento de la señal total ocurre el digito.
    t_seg   = (0:length(seg)-1) / fm;
    subplot(7, 1, k);
    plot(t_seg, seg);
    title(sprintf('D%d  |  %.3f - %.3f s  |  dur = %.3f s', ...
          k, inicios_m(k)/fm, fines_m(k)/fm, length(seg)/fm));
    ylabel('amp'); ylim([-2 2]);
end
xlabel('t[s]');

## -------------------------------------------------------
## PASO 7: Identificación por producto interno
## -------------------------------------------------------

## Cada digito DTMF es la suma de exactamente dos senoidales:
## una de fila y una de columna. Si calculo el producto interno
## de la señal contra cada senoidal de referencia, la que
## corresponde al digito va a dar un valor mucho mas alto que las demás,
## porque las senoidales de distintas frecuencias son ortogonales entre si.

## Por que uso sin Y cos (y no solo sin):
##   La señal grabada puede tener cualquier fase φ:
##   x(t) = A·sin(2πft + φ)
##   Si φ=90°, la señal es practicamente un coseno, y el producto
##   interno con seno daría cercano a cero aunque la frecuencia sea
##   correcta. Para evitar esto, proyecto sobre sin y cos por separado
##   y calculo la magnitud del resultado:
##   |PI(f)| = sqrt( <x,sin>² + <x,cos>² )
##   Esto captura toda la energía en la frecuencia f sin importar la fase.
##   Es equivalente a calcular la magnitud espectral en esa frecuencia.

## Por que uso solo el 50% central del segmento:
##   Al inicio del tono hay un transitorio donde la senoidal todavia
##   no está completamente establecida, y al final el ruido ya domina
##   antes de que se corte formalmente el segmento.
##   El centro del segmento tiene el tono estable y limpio.
##   Se verifico con el digito 1: con el segmento completo ganaba
##   770Hz con valor 22.5 (incorrecto). Con el centro ganaba
##   697Hz con valor 49.6, con mucho margen (correcto).
##   El 25% de margen en cada extremo es suficiente para evitar
##   los transitorios sin desperdiciar demasiada señal util.

freq_filas    = [697, 770, 852, 941];
freq_columnas = [1209, 1336, 1477];
teclado = {'1','2','3'; '4','5','6'; '7','8','9'; '*','0','#'};

fprintf('\n=== Resultado ===\n');
numero = '';

for k = 1:length(inicios)
    seg_completo = x(inicios_m(k):fines_m(k));
    N = length(seg_completo);

    ## Tomar el 50% central: descartar el primer y ultimo 25%
    margen   = round(N * 0.25);
    segmento = seg_completo(margen : N-margen);

    ## Eje de tiempo local para el segmento central, en segundos.
    ## Necesito que arranque en 0 para que sin(2*pi*f*t_seg) tenga
    ## la fase correcta relativa al segmento.
    t_seg = (0:length(segmento)-1)' / fm;   ## vector columna para que dot funcione bien

    ## Producto interno con cada frecuencia de fila
    pi_filas = zeros(1, length(freq_filas));
    for i = 1:length(freq_filas)
        s = sin(2*pi*freq_filas(i)*t_seg);
        c = cos(2*pi*freq_filas(i)*t_seg);
        ## Magnitud de la proyección sobre el subespacio {sin, cos}
        ## de la frecuencia freq_filas(i)
        pi_filas(i) = sqrt(dot(segmento,s)^2 + dot(segmento,c)^2);
    end

    ## Idem para frecuencias de columna
    pi_cols = zeros(1, length(freq_columnas));
    for j = 1:length(freq_columnas)
        s = sin(2*pi*freq_columnas(j)*t_seg);
        c = cos(2*pi*freq_columnas(j)*t_seg);
        pi_cols(j) = sqrt(dot(segmento,s)^2 + dot(segmento,c)^2);
    end

    ## La frecuencia ganadora es la que tiene mayor producto interno.
    ## max() devuelve el valor maximo y su indice.
    ## El indice sirve para buscar la frecuencia y el digito en las tablas.
    [val_f, idx_f] = max(pi_filas);
    [val_c, idx_c] = max(pi_cols);
    digito = teclado{idx_f, idx_c};
    numero = [numero, digito];

    fprintf('D%d: %d Hz (%.1f)  +  %d Hz (%.1f)  ->  %s\n', ...
            k, freq_filas(idx_f), val_f, freq_columnas(idx_c), val_c, digito);
end

fprintf('\nNúmero marcado: %s\n', numero);
